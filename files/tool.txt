def decrypt(key, encrypted):
    shift = len(key)
    decrypted = ""
    for c in encrypted:
        if c.isalpha():
            base = ord('a') if c.islower() else ord('A')
            decrypted += chr((ord(c) - base - shift) % 26 + base)
        else:
            decrypted += c
    return decrypted
